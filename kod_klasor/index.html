								public function getir($tablo, $siralanacak_alan="", $siralama_sekli="", $kac_tane=0, $baslangic=0)
								{
									if ($siralanacak_alan=="")
									{
										if ($kac_tane==0)
											return $this->db->get($tablo)->result_array();
										else
											return $this->db->get($tablo,$kac_tane,$baslangic)->result_array();
									}
									else
									{
										if ($kac_tane==0)
											return $this->db->order_by($siralanacak_alan,$siralama_sekli)->get($tablo)->result_array();
										else
											return $this->db->order_by($siralanacak_alan,$siralama_sekli)->get($tablo,$kac_tane,$baslangic)->result_array();
									}
								}
